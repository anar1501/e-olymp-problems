#include <stdio.h>
#include <stdlib.h>

int main()
{
    int n,m;
    scanf("%d %d",&n,&m);
    int arr[n][m];
    for(int i=0;i<n;i++){
        for(int j=0;j<m;j++){
            scanf("%d",&arr[i][j]);
        }
    }
    printf("%d %d\n",m,n);
    for(int j=0;j<m;j++){
        for(int i=n-1;i>=0;i--){
        printf("%d ",arr[i][j]);
        }
        printf("\n");
    }



    return 0;
}
